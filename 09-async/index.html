Последовательное выполнение


function foo() {
    console.log('функция foo была вызвана');
}
function bar() {
    console.log('функция bar была вызвана')
}

foo();
bar();


> функция foo была вызвана
> функция bar была вызвана

Асинхронное выполнение


function foo() {
    console.log('функция foo была вызвана');
}
function bar() {
    console.log('функция bar была вызвана')
}

setTimeout(function () {
    foo();
}, 1000);
bar();

> функция bar была вызвана
> функция foo была вызвана
