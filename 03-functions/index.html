function (x, y) {
    var sum = x + y;

    if (arguments[2]) {
        sum += arguments[2];
    }

    return sum;
}

array-like


function addMany() {
    var sum = 0;
    var length = arguments.length;

    for (var i = 0; i < length; i += 1) {
        sum += arguments[i];
    }

    return sum;
}
            

addMany(2, 3, 4);   // 9
