function myMin(a, b) {
    if (b === undefined) {
        return a;
    }

    return a < b ? a : b;
}
            

myMin(13);      // 13

Значение по умолчанию


function myMin(a, b) {
    b = b || Infinity;

    return a < b ? a : b;
}
            

myMin(2, 7);    // 2
myMin(13, 7);   // 7
myMin(-13);     // -13

function myMin(a, b) {
    var min = a < b ? a : b;
    var c = arguments[2];

    return c < min ? c : min;
}

array-like


function addMany() {
    var sum = 0;
    var length = arguments.length;

    for (var i = 0; i < length; i += 1) {
        sum += arguments[i];
    }

    return sum;
}
            

addMany(2, 3, 4);   // 9
